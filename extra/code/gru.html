#8 timeframes, 400 dps per timeframe

inputs = tf.keras.Input(shape=(X.shape[1],))
x = tf.keras.layers.Dense(128, activation='relu')(inputs)
x = tf.keras.layers.RepeatVector(8)(x) 
x = tf.keras.layers.GRU(128, return_sequences=True)(x)
x = tf.keras.layers.GRU(128, return_sequences=True)(x)

#apply dense on each of the timeframes
outputs = tf.keras.layers.TimeDistributed(tf.keras.layers.Dense(400))(x)
In[169]:
